微软历史上出现安全问题最多的协议
实现复杂

SMB扫描

nmap–SMB扫描
nmap -v -p139,445 192.168.254.0/24
nmap 192.168.254.102 -p139,445 --script=smb-os-discovery.nse #判断系统信息
nmap -v -p139,445 --script=smb-check-vulns --script-args=unsafe=1 192.168.254.102 #探测漏洞信息
ls /usr/share/nmap/scripts/  #nmap脚本放置路径
nbtscan–SMB扫描
nbtscan -r 192.168.254.0/24
enum4linux–SMB扫描
enum4linux -a 192.168.254.102

SMTP扫描

nc–SMTP扫描
nc -nc 1.1.1.1 25
HELO 163.com    
VRFY root
nmap–SMTP扫描
nmap smtp.163.com -p25 --script=smtp-enum-users.nse --script-args=smtp-enum-users.methods={VRFY}
nmap smtp.163.com -p25 --script=smtp-open-relay.nse
smtp-user-enum–SMTP扫描
smtp-user-enum -M VRFY -U users.txt -t 10.0.0.1
python ./smtp_scan.py
